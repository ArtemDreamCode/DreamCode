		<div v-for="device in devices">
			<div v-if = "device.pinged">
				<div v-if = "device.state">
				   <button class = "btn_on" v-on:click="relay({ip: device.ip, turn: device.state ? 'off': 'on'})">{{ device.state ? 'off': 'on' }} </button>
				 </div>
				 <div v-else>
				   <button class = "btn_off" v-on:click="relay({ip: device.ip, turn: device.state ? 'off': 'on'})">{{ device.state ? 'off': 'on' }} </button>

	<script type="text/javascript">
	
	//	const socket = io("ws://172.20.10.2:3000", { transports: ["websocket"] });		
		const socket = io("ws://172.20.10.11:3000", { transports: ["websocket"] });
	//	const socket = io("ws://192.168.0.102:3000", { transports: ["websocket"] });
		//const socket = io("ws://raspberrypi:3000", { transports: ["websocket"] });	
		socket.on("devices", devices => {
			//console.log(devices)
			app.devices = devices
		})
		
		let app = new Vue({
		  el: '#app',
		  data: {
		    devices: new Map()
		  },
		  mounted: function () {
		  	socket.emit("deviceReq", (devs) => {
				app.devices = devs
		  	})
		  },
		  methods: {
		  	relay(data) {
		  		socket.emit("relay", (data))// send data  				  		
		  		
		  	}
		  }
		})
		
	</script>
