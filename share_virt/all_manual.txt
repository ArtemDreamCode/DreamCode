5. PostgreSQL
	5.1 Установка и настройка
	    [sudo apt-get --purge remove postgresql postgresql-doc postgresql-common (полное удаление pg)]
		sudo apt install postgresql
		sudo apt install postgresql-server-dev-12
		sudo apt-get -y install postgresql-12-cron
		sudo systemctl is-active postgresql (активность службы)
		sudo systemctl is-enabled postgresql (включена ли служба)
		sudo systemctl status postgresql (статус службы)
		sudo pg_isready (pg сервер готовность принимать подключения)
		sudo systemctl restart postgresql (рестарт службы)
	5.2 Создание бд
		sudo su - postgres (подключение к системе с помощью учетной записи postgres)
			psql
			    [DROP USER artem;] (удаление юзера/роли)
				CREATE USER artem WITH PASSWORD '1';
				ALTER ROLE artem SUPERUSER;
				CREATE DATABASE devdb;
				\list (список баз)
				GRANT ALL PRIVILEGES ON DATABASE devdb to artem;
				\q
			exit
	5.3 Настройка клиентской аутентификации
		5.3.1 postgresql.conf
			sudo nano /etc/postgresql/12/main/postgresql.conf
			Раскомментируем или добавим строку: listen_addresses = '*'
			Мы разрешили прослушивание запросов от всех IP-адресов.
		-- for pgcron--
		   sudo nano /etc/postgresql/12/main/postgresql.conf
		   ниже строчки # - Shared Library Preloading -
			вставляем:
				shared_preload_libraries = 'pg_cron'    # (change requires restart)
				cron.database_name = 'devdb'
		5.3.2 pg_hba.conf
			sudo nano /etc/postgresql/12/main/pg_hba.conf
			Добавляем запись в секцию # IPv4 local connections:
			host     devdb     all     192.168.../32     md5
			Запись разрешает подключение к БД mybd всем пользователям с IP адресом 192.168..., используя пароль.
		--for pgcron--
			добавляем строки
				# Schedule jobs via background workers instead of localhost connections
				cron.use_background_workers = on
				# Increase the number of available background workers from the default of 8
				max_worker_processes = 20
				# Connect via a unix domain socket
				cron.host = '/tmp'
		5.3.3 sudo systemctl restart postgresql
	5.4 pgAdmin4 (ставим на хосте)
		5.4.1 Создаем новое подключение к серверу: Add New Server
		  Вкладка General:
		    Name: название сервера (произвольное) devdb;
		  Вкладка Connection:
			Host Name: RemoteServerIP;
			Port: 5432;
			Maintenance database: devdb;
			Username: artem	
			Password: 1
